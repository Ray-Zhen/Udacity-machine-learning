def performance_metric2(y_true, y_predict):
    """"""
    import numpy as np
    y_true = np.array(y_true)
    y_predict = np.array(y_predict)
    
    y_mean = sum(y_true)/len(y_true)
    SS_tot = sum((y_true - y_mean)**2)
    SS_reg = sum((y_predict - y_mean)**2)
    SS_res = sum((y_true - y_predict)**2)
    r2 = 1 - SS_res/SS_tot
    
    score = r2

    return score

    """  [X,y]"""
    from sklearn.model_selection import KFold, GridSearchCV
    from sklearn.tree import  DecisionTreeRegressor
    from sklearn.metrics import make_scorer
    
    cross_validator = KFold(n_splits=10, random_state=42)

    params = {'max_depth':range(1,11)}

    grid = grid.fit(X, y)

def cross_val_score(estimator, X, y, scoring = performance_metric, cv=3):
    """  """
    from sklearn.model_selection import KFold
    
    kf = KFold(n_splits=cv)
    scores = []   
    for train_index, test_index in kf.split(X):
        X_train, X_test = np.array(X)[train_index], np.array(X)[test_index]
        y_train, y_test = np.array(y)[train_index], np.array(y)[test_index]
        reg = estimator.fit(X_train, y_train)
        y_predict = reg.predict(X_test)
        scores.append( performance_metric(y_test, y_predict) )
#    scores = [0,0,0]
    return scores


def fit_model2(X, y):
    """  [X,y]"""
    from sklearn.tree import  DecisionTreeRegressor
    
    best_score = 0
    max_depth = range(1,11)
    
#   max_depth 
    for i in max_depth:
        regressor = DecisionTreeRegressor(max_depth = i, random_state=0)
        scores = cross_val_score(estimator=regressor, X=X, y=y, scoring = performance_metric, cv=3)
        score_mean = np.array(scores).mean()  #
#       
        if score_mean > best_score:
            best_score = score_mean
            best_estimator = regressor
            
    #
  #  best_estimator = None
    return best_estimator
